Elige el tipo de dispositivo de red

R1>enable
R1#configure terminal


!-- Configuramos la interfaz Fa0/1
R1(config)#interface fa0/1
R1(config-if)#ip address 192.168.10.1 255.255.255.0
R1(config-if)#no shutdown
R1(config-if)#exit


!-- Creamos un pool DHCP para brindar direccionamiento a nuestros equipos en la red
!-- Tambien activamos la opcion 150
R1(config)#ip dhcp pool LAN_A
R1(dhcp-config)#network 192.168.10.0 255.255.255.0
R1(dhcp-config)#default-router 192.168.10.1
R1(dhcp-config)#dns 8.8.8.8
R1(dhcp-config)#option 150 ip 192.168.10.1
R1(dhcp-config)#exit
R1(config)#ip dhcp excluded-address 192.168.10.1


!-- Configuramos el servicio telefonico CME
R1(config)#telephony-service
R1(config-telephony)#max-dn 3
R1(config-telephony)#max-ephones 3
R1(config-telephony)#ip source-address 192.168.10.1 port 2010
R1(config-telephony)#auto assign 1 to 3


!-- Configuramos el numero de extension de los telefonos
R1(config-telephony)#ephone-dn 1
R1(config-ephone-dn)#number 1010

R1(config-ephone-dn)#ephone-dn 2
R1(config-ephone-dn)#number 1020

R1(config-ephone-dn)#ephone-dn 3
R1(config-ephone-dn)#number 1030

S1>enable
S1#configure terminal


!--Configuramos la VLAN 1 en los puertos Fa0/2 - Fa0/3
!-- para que funcione como una VLAN de voz
S1(config)#interface range fa0/2-3
S1(config-if)#switchport mode access
S1(config-if)#switchport voice vlan 1
S1(config-if)#exit

!-- En el puerto Fa0/4, deberemos configurar la VLAN 1
!-- como VLAN de voz y de datos
S1(config)#interface fa0/4
S1(config-if)#switchport mode access
S1(config-if)#switchport access vlan 1 #VLAN de datos para la PC
S1(config-if)#switchport voice vlan 1 #VLAN de voz para VoIP 3


!-- No es necesario configurar un enlace troncal en Fa0/1
!-- ya que solo estamos trabajando con la VLAN por defecto (VLAN 1)

enable
configure terminal

interface range fa0/2-3
switchport mode access
switchport voice vlan 1
exit

interface fa0/4
switchport mode access
switchport access vlan 1
switchport voice vlan 1
